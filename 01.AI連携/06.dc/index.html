    Chirimen(DC Motor)

    sketch.js

    DC Motorを操作します。

    I2Cの認識状況の確認には I2C Detect を用います。今回は、PCA9685を使用するので0x40が認識できていれば動作します。

    FaBo ShieldのDCジャックに5Vを供給してください。5Vの電源供給しないとI2Cの認識ができません。

    DCの値は、0x06〜0x3Fの値が設定できます。FORWARDが前進, BACKWARDが後進です。

/* ===
ml5 Example
Webcam Image Classification using a pre-trained customized model and p5.js
This example uses p5 preload function to create the classifier
=== */

// Classifier Variable
let classifier;
// Model URL
let imageModelURL = 'https://teachablemachine.withgoogle.com/models/oF8lUO4C/model.json';

// Video
let video;
let flippedVideo;
// To store the classification
let label = "";

let port4;
let port5;

let pca9685;
let motor = {};

let FORWARD = 1;
let BACK = 2;

// Load the model first
function preload() {
  classifier = ml5.imageClassifier(imageModelURL);
}

async function init() {
  var gpioAccess = await navigator.requestGPIOAccess(); 
  port4 = gpioAccess.ports.get(4);
  port5 = gpioAccess.ports.get(5);
  await port4.export("out");
  await port5.export("out");

  var i2cAccess = await navigator.requestI2CAccess();
  var port = i2cAccess.ports.get(1);
  pca9685 = new PCA9685(port, 0x40);
  await pca9685.init(0.001, 0.002, 30);

  // motor
  motor[0] = await port.open(0x63);
  motor[1] = await port.open(0x64);
  await motor[0].write8(0x00,0x00);
  await motor[1].write8(0x00,0x00);  
}


function setup() {
   init();

  createCanvas(320, 260);
  // Create the video
  video = createCapture(VIDEO);
  video.size(320, 240);
  video.hide();

  flippedVideo = ml5.flipImage(video)
  // Start classifying
  classifyVideo();
}

function draw() {
  background(0);
  // Draw the video
  image(flippedVideo, 0, 0);

  // Draw the label
  fill(255);
  textSize(16);
  textAlign(CENTER);
  text(label, width / 2, height - 4);
}

// Get a prediction for the current video frame
function classifyVideo() {
  flippedVideo = ml5.flipImage(video)
  classifier.classify(flippedVideo, gotResult);
}

async function controlServo(num, angle) {
  await pca9685.setServo(num, angle);
}

async function controlMotor(num, direction, speed) {
  let value = (speed<<2) + direction + 1&0xff;
  await motor[num].write8(0x00,value); 
}

// When we get a result
function gotResult(error, results) {
  // If there is an error
  if (error) {
    console.error(error);
    return;
  }
  // The results are in an array ordered by confidence.
  // console.log(results[0]);
  label = results[0].label;

  if(label == "Class 1") {
    port4.write(0);
    port5.write(1);
    controlServo(0, 30);
    // Speed: 0x06-0x3F
    controlMotor(0,FORWARD,0x3F);
  } else if(label == "Class 2") {
    port4.write(1);
    port5.write(0);
    controlServo(0, -30);
    // Speed: 0x06-0x3F
    controlMotor(0,BACK,0x3F);
  }

  // Classifiy again!
  classifyVideo();
}
